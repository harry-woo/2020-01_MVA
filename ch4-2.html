성능 평가를 위하여 fit 결과를 분류

z_iris3 <- z_iris2 %>% 
  mutate(fit = ifelse(fit == 2, "setosa", ifelse(fit == 3, "versicolor", "virginica")))
confu_mat2 <- table(z_iris3$Species, z_iris3$fit)
